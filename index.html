<!DOCTYPE html>

<html>

<head>

    <title>Reading Experiment</title>

    <!-- Load jsPsych -->

    <script src="https://unpkg.com/@jspsych-contrib/plugin-pipe"></script>

    <link href="https://unpkg.com/css/jspsych.css" rel="stylesheet" type="text/css">

</head>

<body></body>

<script>

    // Initialize jsPsych

    const jsPsych = initJsPsych({

        on_finish: function() {

            // DataPipe: Send data to OSF

            jsPsych.data.get().localSave('csv', 'experiment_data.csv');

        }

    });


    const timeline = [];


    // 1. Participant Subject Assignment (DataPipe)

    // DataPipe automatically adds 'condition' to the URL

    // Here we assume 2 conditions: 0 and 1

    const urlParams = new URLSearchParams(window.location.search);

    const condition = urlParams.get('condition') || 0; // Default to 0 if not running on DataPipe


    // 2. Define text based on condition

    let experimentText = "";

    if (condition == 0) {

        experimentText = "<h1>Condition A</h1><p>This is the text for condition A. Read it carefully.</p>";

    } else {

        experimentText = "<h1>Condition B</h1><p>This is the text for condition B. Read it carefully.</p>";

    }


    // 3. Welcome/Instructions

    timeline.push({

        type: jsPsychInstructions,

        pages: ['Welcome.', 'Read the text and press SPACEBAR to answer how you feel.', 'Press any key to begin.'],

        show_clickable_nav: true

    });


    // 4. Main Experiment Loop

    let spacePressCount = 0;

    

    // Read Screen Trial

    let readTrial = {

        type: jsPsychHtmlKeyboardResponse,

        stimulus: experimentText + "<br><p><strong>Press SPACEBAR</strong></p>",

        choices: [' '], // Only allow spacebar

        on_finish: function(data) {

            // Increment the counter

            spacePressCount++;

            data.space_press_count = spacePressCount; // Record in data

        }

    };


    // Question Trial

    let questionTrial = {

        type: jsPsychSurveyText,

        questions: [{prompt: 'How do you feel right now?', rows: 1}],

        on_finish: function(data) {

            // Append the total count to this trial's data

            data.total_space_presses = spacePressCount;

            data.condition = condition;

        }

    };


    // Loop the read+question block 5 times

    let experimentLoop = {

        timeline: [readTrial, questionTrial],

        repetitions: 5

    };

    timeline.push(experimentLoop);


              const subject_id = jsPsych.randomization.randomID(10);

              const filename = `${subject_id}.csv`;


              const save_data = {

                type: jsPsychPipe,

                action: "save",

                experiment_id: "JX4hJtjy26fD",

                filename: filename,

                data_string: ()=>jsPsych.data.get().csv()

              };


    // 5. Run the experiment

    jsPsych.run(timeline);

</script>

</html>
